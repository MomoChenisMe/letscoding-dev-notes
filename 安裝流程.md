1. 架設Linode主機
2. ssh連線到主機
3. apt update && apt install curl -y
4. curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.40.4/install.sh | bash
5. source ~/.bashrc
6. nvm -v
7. nvm install --lts
8. node -v
9. apt update && apt install git -y
10. npm install -g @google/gemini-cli（選用）
11. npm i -g openclaw@latest
12. openclaw onboard --install-daemon
===（選用）===
13. mkdir -p ~/.config/systemd/user
14. nano ~/.config/systemd/user/openclaw-gateway.service
15. 確認是否有以下內容：
```
[Unit]
Description=OpenClaw Gateway (profile: <profile>, v<version>)
After=network-online.target
Wants=network-online.target

[Service]
ExecStart=/usr/local/bin/openclaw gateway --port 18789
Restart=always
RestartSec=5

[Install]
WantedBy=default.target
```
16. systemctl --user daemon-reload
17. systemctl --user enable --now openclaw-gateway.service
18. systemctl --user status openclaw-gateway.service（查看服務狀態）
======
19. wget https://dl.google.com/linux/direct/google-chrome-stable_current_amd64.deb
20. sudo dpkg -i google-chrome-stable_current_amd64.deb
21. sudo apt -f install -y
22. google-chrome-stable --version
23. apt install -y   libnss3   libatk-bridge2.0-0   libdrm2   libxkbcommon0   libxcomposite1   libxdamage1   libxrandr2   libgbm1   libasound2
24. apt install -y   libnss3   libatk-bridge2.0-0t64   libdrm2   libxkbcommon0   libxcomposite1   libxdamage1   libxrandr2   libgbm1   libasound2t64
25. npm install -g playwright
26. npx playwright install --with-deps chromium
27. npx playwright --version
28. 編輯 ~/.openclaw/openclaw.json
```
{
  "browser": {
    "enabled": true,
    "executablePath": "/usr/bin/google-chrome-stable",
    "headless": true,
    "noSandbox": true
  }
}
```
